    # onClick (optional)

    Called when the user clicks a chessboard cell.

    Returns the clicked cell data and its position on the board.
    <br/>
    <h2 id="Typing" data-anchor>Typing</h2>
    <CopyBlock client:load>
        ```ts
        type ClickData = {
            cellData: {
                figure: { 
                    color: "white" | "black";
                    type: "pawn" | "bishop" | "knigts" | "rook" | "queen" | "king";
                } | undefined;
            };
            pos: [number, number];
        }

        type onClick = (data: ClickData) => void;
        ```
    </CopyBlock>

    <br />
    <h2 id="Code" data-anchor>Code</h2>
    <CopyBlock client:load>
