# Get started

## Install
```bash
npm install react-chessboard-ui
```
<br />
## Add in project
```tsx

import { ChessBoard } from "react-chessboard-ui";

// Import chessboard styles global or local
import 'react-chessboard-ui/dist/index.css';

    return (
        <div className="app">

            />
        </div>
    );
}
```
